## Selecting behaviour using cases

a="dog"
case $a in 
    cat) echo "Feline";;
    dog|puppy) echo "Canine";;
    *) echo "No match!";;
esac

## Using functions

function numberthings {
    i=1
    for f in $@; do
        echo $i: $f
        ((i+=1))
    done
}

numberthings $(ls)
numberthings pine birch maple syrup
